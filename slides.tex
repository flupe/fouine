\begin{frame}[fragile]
\frametitle{Types somme}
\begin{minted}{ocaml}
>>> type ('a, 'b) pair = Left of 'a | Right of 'b;;
>>> type 'a tree = Leaf | Node of 'a * 'a tree * 'a tree;;
>>> Leaf;;
- : '_a tree = Leaf
\end{minted}
\begin{minted}{ocaml}
>>> type 'a list = [] | (::) of 'a * 'a list;;
>>> [1; 2; 3]
- : int list = (::) (1, (::) (2, (::) (3, [])))
\end{minted}
